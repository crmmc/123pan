    **Step 1: 修改 _create_directory 以暴露 429 信息**

    当前 `_create_directory`（第 484-514 行）在遇到任何错误时都返回 None。为了让 wrapper 能区分 429 和其他错误，需要做以下调整：

    将 `_create_directory` 中的 `self.session.post(...)` 替换为 `self._api_request(self.session.post, ...)`（如果 Plan 01 尚未完成此迁移）。这样 _api_request 会先处理 token 过期。

    然后，在 `_create_directory` 中，在 `res_mk = ...` 之后、解析 json 之前，添加 429 检测并抛出特定异常：

    ```python
    def _create_directory(self, parent_id, dirname):
        """在指定父目录下创建文件夹。

        Returns:
            新创建的目录 ID

        Raises:
            RateLimitError: 收到 HTTP 429 响应
            RuntimeError: 其他业务错误
        """
        url = "https://www.123pan.com/a/api/file/upload_request"
        data_mk = {
            "driveId": 0,
            "etag": "",
            "fileName": dirname,
            "parentFileId": parent_id,
            "size": 0,
            "type": 1,
            "duplicate": 1,
            "NotReuse": True,
            "event": "newCreateFolder",
            "operateType": 1,
        }
        res_mk = self._api_request(
            self.session.post, url,
            headers=self.header_logined,
            data=json.dumps(data_mk),
            timeout=10,
        )
        if res_mk.status_code == 429:
            raise RateLimitError(f"创建目录 '{dirname}' 触发 429 限流")
        try:
            res_json = res_mk.json()
        except json.decoder.JSONDecodeError:
            logger.error("创建失败")
            logger.error(res_mk.text)
            raise RuntimeError(f"创建目录 '{dirname}' 响应解析失败")
        code_mkdir = res_json.get("code", -1)

        if code_mkdir == 0:
            logger.info(f"创建成功: {res_json['data']['Info']['FileId']}")
            return res_json["data"]["Info"]["FileId"]
        logger.error(f"创建失败: {res_json}")
        raise RuntimeError(
            f"创建目录 '{dirname}' 失败: code={code_mkdir}, message={res_json.get('message', '')}"
        )
    ```

    注意：原 `_create_directory` 在失败时返回 None，调用方通过 `if not result` 检查。改为抛异常后，调用方的 `if not` 检查永远不会再触发，但保留它们作为防御性代码不会出错。

    **Step 3: 添加 _create_directory_with_backoff 作为 wrapper**

    在 Pan123 类中添加（放在 `_create_directory` 之后）：

    ```python
    # 退避常量
    _INITIAL_BACKOFF = 1.0
    _MAX_BACKOFF = 30.0
    _BACKOFF_MULTIPLIER = 2
    _MAX_429_RETRIES = 10

    def _create_directory_with_backoff(self, parent_id, dirname):
        """创建目录，遇到 429 自适应指数退避。

        复用 _create_directory 的核心请求逻辑，只在循环中添加退避和重试。
        """
        backoff = self._INITIAL_BACKOFF
        last_error = None
        for attempt in range(self._MAX_429_RETRIES + 1):
            try:
                return self._create_directory(parent_id, dirname)
            except RateLimitError as e:
                last_error = e
                if attempt >= self._MAX_429_RETRIES:
                    raise RuntimeError(
                        f"创建目录 '{dirname}' 触发 429 限流，已重试 {self._MAX_429_RETRIES} 次"
                    ) from e
                logger.warning(
                    "目录创建 429，退避 %.1fs（第 %d 次）",
                    backoff, attempt + 1,
                )
                time.sleep(backoff)
                backoff = min(backoff * self._BACKOFF_MULTIPLIER, self._MAX_BACKOFF)
            # RuntimeError 和其他异常直接向上抛出，不重试
        raise RuntimeError(
            f"创建目录 '{dirname}' 超出最大重试次数"
        ) from last_error
    ```

    **核心设计：** wrapper 只负责退避循环，所有请求构造、发送、解析逻辑都在 `_create_directory` 中。`_create_directory` 抛 `RateLimitError` 表示 429（需要重试），抛 `RuntimeError` 表示业务错误（直接终止）。wrapper 捕获 `RateLimitError` 进行退避重试，遇到 `RuntimeError` 直接向上传播。

    **Step 1: 修改 PrepareUploadTask.run 添加文件级容错**

    当前 `PrepareUploadTask.run`（file_interface.py 第 472-507 行）中，文件处理部分（第 496-503 行）没有 try/except。需要为单个文件的处理添加异常捕获：

    ```python
    def run(self):
        try:
            uploads = []
            created_dir_count = 0
            folder_items = []
            for local_path in self.local_paths:
                path = Path(local_path)

                if path.is_dir():
                    # 目录上传 — 失败直接抛异常终止整个操作
                    plan = self.pan.prepare_folder_upload(
                        path, self.target_dir_id
                    )
                    uploads.extend(plan["file_targets"])
                    created_dir_count += int(plan["created_dir_count"])
                    folder_items.append({
                        "FileId": plan["root_dir_id"],
                        "FileName": plan["root_dir_name"],
                    })
                    continue

                # 文件上传 — 单个文件失败只跳过该文件
                try:
                    if not path.exists():
                        logger.warning("文件不存在，跳过: %s", path)
                        continue
                    uploads.append({
                        "file_name": path.name,
                        "file_size": path.stat().st_size,
                        "local_path": str(path),
                        "target_dir_id": self.target_dir_id,
                    })
                except (OSError, PermissionError) as e:
                    logger.warning("文件处理失败，跳过: %s (%s)", path, e)
                    continue

            self.signals.finished.emit(uploads, created_dir_count, folder_items, "")
        except Exception as e:
            self.signals.finished.emit([], 0, [], str(e))
    ```

    关键变化：
    1. 目录处理（`path.is_dir()`）: 保持无 try/except — prepare_folder_upload 失败会抛异常，被外层 except 捕获，整个操作中止
    2. 文件处理: 添加 try/except，单个文件失败跳过，记录 warning 日志
    3. `path.exists()` 检查: 对不存在的文件提前跳过
    4. `path.stat()` 的 OSError/PermissionError: 捕获后跳过
